  def one_of(*variants)
    return if variants.empty?
    err = nil
    variants.each do |v|
      begin
        cur_index = @current_atom_index
        i(v)
      rescue SyntaxError => e
        err = e
        @current_atom_index = cur_index
      else
        return
      end
    end
    raise err
  end
