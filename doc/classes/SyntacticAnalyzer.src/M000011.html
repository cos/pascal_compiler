  def analize
    @current_atom_index = 0
    begin
      program_sursa
    rescue SyntaxError
      s = deepest_errors.inject('') {|s,e| s+= ' or ' +e.expected}
      raise 'Expected '+s+', found '+deepest_errors[0].actual+" on line "+deepest_errors[0].line.to_s
    end
  end
